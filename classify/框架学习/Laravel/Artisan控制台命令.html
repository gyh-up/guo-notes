artisan 控制台 #

简介 #

[Laravel Sail](Sail 开发环境 | 官方扩展包 |《Laravel 9 中文文档 9.x》| Laravel China 社区 (learnku.com)) 是一个轻量级的命令行界面，用于与 Laravel 的默认 Docker 开发环境进行交互。 Laravel Tinker 是为 Laravel 提供的强大的 REPL（交互式解释器），由 PsySH 提供支持。

常用命令 #

shell
#命令列表
php artisan list 
#查看命令帮助界面
php artisan help SendEmails 
#生成命令
php artisan make:command SendEmails

定义输入期望 #

php
// ------------- 参数 -------------
//必需的参数
protected $signature = 'mail:send {user}';
// 可选参数...
protected $signature = 'mail:send {user?}'
// 带有默认值的可选参数...
protected $signature = 'mail:send {user=foo}'
// ------------- 选项 -------------    
// 选项，如果传递了 --queue 选项，该选项的值将会是 true。否则其值为 false
protected $signature = 'mail:send {user} {--queue}';
// 带值的选项，为一个选项指定一个值，未指定该选项，则其值为 null
protected $signature = 'mail:send {user} {--queue=}';
protected $signature = 'mail:send {user} {--queue=default}';
// 选项简写，php artisan mail:send 1 -Q，前缀只用一个连字符
protected $signature = 'mail:send {user} {--Q|queue}';
// ------------- 输入数组 -------------
// 输入数组，用 * 号指定一个数组参数
protected $signature = 'mail:send {user*}';
// 选项数组，php artisan mail:send --id=1 --id=2
protected $signature ='mail:send {user} {--id=*}';
// ------------- 输入说明 -------------
// 输入说明，可以自由地定义在多行中
protected $signature = 'mail:send
                        {user : The ID of the user}
                        {--queue : the job should be queued}';

命令 I/O #

php
// ------------- 检索输入 -------------
// 检索输入参数
$userId = $this->argument('user');
// 检索所有的参数
$arguments = $this->arguments();
// 检索输入选项
$queueName = $this->option('queue');
// 检索所有的选项
$options = $this->options();
// ------------- 交互式输入 -------------
// 询问用户指定的问题来接收用户输入 ask
$name = $this->ask('What is your name?');
// secret 方法与 ask 相似，区别在于用户的输入将不可见
$password = $this->secret('What is the password?');
// 请求确认，默认返回 false
if ($this->confirm('Do you wish to continue?')) {
    // 默认返回 false。如果用户输入 y 或 yes，将返回 true
}
// 请求确认，默认返回 true
if ($this->confirm('Do you wish to continue?', true)) {
    // 默认返回 true。
}
// 自动补全
$name = $this->anticipate('What is your name?', ['Taylor', 'Dayle']);
// 自动补全，闭包作为第二个参数
$name = $this->anticipate('What is your address?', function ($input) {
    // 可以将一个闭包作为第二个参数，如查询数据库
});
// 多选择问题，当询问问题时，如果需要给用户一个预定义的选择，可以使用 choice 方法。如果没有选项被选择，您可以设置数组索引的默认值去返回，通过这个方法的第三个参数去传入索引，第四和第五可选参数 ，用于确定选择有效响应的最大尝试次数以及是否允许多次选择
$name = $this->choice(
    'What is your name?',
    ['Taylor', 'Dayle'],
    $defaultIndex,
    $maxAttempts = null,
    $allowMultipleSelections = false
);
// ------------- 文字输出 -------------
// 可以使用 line，info，comment，question 和 error 方法，发送输出到控制台。 这些方法中的每一个都会使用合适的 ANSI 颜色以展示不同的用途。
// 绿色文本
$this->info('The command was successful!'); 
// 红色字体
$this->error('Something went wrong!');
// 红色字体
$this->line('Display this on the screen');
// 输出单行空白...
$this->newLine();
// 输出三行空白...
$this->newLine(3);
// 自动表格化，提供表的列名和数据
$this->table(
    ['Name', 'Email'],
    User::all(['name', 'email'])->toArray()
);
// 进度条
$users = $this->withProgressBar(User::all(), function ($user) {
    $this->performTask($user);
});
// 进度条手动控制
$users = App\Models\User::all();
$bar = $this->output->createProgressBar(count($users));
$bar->start();
foreach ($users as $user) {
    $this->performTask($user);
    $bar->advance();
}
$bar->finish();

注册命令 #

您的所有控制台命令都在您的应用程序的 App\Console\Kernel 类中注册，这是您的应用程序的「控制台内核」。在此类的 commands 方法中，您将看到对内核的 load 方法的调用。 load 方法将扫描 app/Console/Commands 目录并自动将其中包含的每个命令注册到 Artisan。 您甚至可以自由地调用 load 方法来扫描其他目录以查找 Artisan 命令：

php
/**
 * 注册应用程序的命令。
 *
 * @return void
 */
protected function commands()
{
    $this->load(__DIR__.'/Commands');
    $this->load(__DIR__.'/../Domain/Orders/Commands');
    // ...
}

以编程方式执行命令 #

有时您可能希望在 CLI 之外执行 Artisan 命令。例如，您可能希望从路由或控制器执行 Artisan 命令。您可以使用 Artisan 外观上的 call 方法来完成此操作。 call 方法接受命令的签名名称或类名作为其第一个参数，以及一个命令参数数组作为第二个参数。将返回退出代码：

php
use Illuminate\Support\Facades\Artisan;

Route::post('/user/{user}/mail', function ($user) {
    $exitCode = Artisan::call('mail:send', [
        'user' => $user, '--queue' => 'default'
    ]);
    //
});
//可以将整个 Artisan 命令作为字符串传递给 `call` 方法
Route::post('/user/{user}/mail', function ($user) {
    $exitCode = Artisan::call('mail:send 1 --queue=default');
    //
});
// 传递数组值
Route::post('/mail', function () {
    $exitCode = Artisan::call('mail:send', [
        '--id' => [5, 13]
    ]);
});
// 传递布尔值
$exitCode = Artisan::call('migrate:refresh', [
    '--force' => true,
]);
// 队列 Artisan 命令，使用 Artisan 门面的 queue 方法，您甚至可以对 Artisan 命令进行排队，以便您的 队列工作者 在后台处理它们。 在使用此方法之前，请确保您已配置队列并正在运行队列侦听器
Route::post('/user/{user}/mail', function ($user) {
    Artisan::queue('mail:send', [
        'user' => $user, '--queue' => 'default'
    ]);
    //
});
// 使用 onConnection 和 onQueue 方法，您可以指定 Artisan 命令应分派到的连接或队列
Route::post('/user/{user}/mail', function ($user) {
    Artisan::queue('mail:send', [
        'user' => 1, '--queue' => 'default'
    ])->onConnection('redis')->onQueue('commands');
    //
});
// ------------- 从其他命令调用命令 -------------
// 正常输出
$this->call('mail:send', [
    'user' => 1, '--queue' => 'default'
]);
// 并禁止其所有输出
$this->callSilently('mail:send', [
    'user' => 1, '--queue' => 'default'
]);

事件 #

Artisan 在运行命令时会调度三个事件： Illuminate\Console\Events\ArtisanStarting，Illuminate\Console\Events\CommandStarting 和 Illuminate\Console\Events\CommandFinished。 当 Artisan 开始运行时，会立即调度 ArtisanStarting 事件。接下来，在命令运行之前立即调度 CommandStarting 事件。最后，一旦命令执行完毕，就会调度 CommandFinished 事件。
